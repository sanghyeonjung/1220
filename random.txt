#include <stdio.h>
#include <stdlib.h>

int main()
{
    for(int k=1; k<=10; k++)
    {
        int randoma = 0; 
        randoma = rand()%20 + 1;
        int randomb = 0; 
        randomb = rand()%20 + 1;
        int pm = rand()%4 + 1;
        int ans = 0;
        char c;
        switch(pm){
            case 1 :
            c = '+'
            ans = randoma + randomb;
            break;
            case 2 :
            c = '-'
            ans = randoma - randomb;
            break;
            case 3 :
            c = '*'
            ans = randoma * randomb;
            break;
            case 4 :
            c = '/'
            ans = randoma / randomb;
            break;
        }
    }
}
